// 分发调度器
const publishers = {
  wechat: new WechatPublisher(config),
  weibo: new WeiboPublisher(config),
  xiaohongshu: new XHSPublisher(config),
};

async function distribute(content) {
  const jobs = Object.entries(publishers).map(async ([platform, pub]) => {
    const adapted = await pub.adapt(content);  // 格式适配
    await queue.add('publish', { platform, data: adapted }, {
      delay: getOptimalTime(platform)  // 最优发布时间
    });
  });
  await Promise.all(jobs);
}
